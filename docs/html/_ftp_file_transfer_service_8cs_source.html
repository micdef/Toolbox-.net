107 public async Task UploadOneAsync(string localPath, string remotePath, bool overwrite = true, CancellationToken cancellationToken = default)
108 {
109 ThrowIfDisposed();
110 ArgumentNullException.ThrowIfNull(localPath);
111 ArgumentNullException.ThrowIfNull(remotePath);
112
113 if (!File.Exists(localPath))
114 {
115 throw new FileNotFoundException("Local file not found.", localPath);
116 }
117
118 cancellationToken.ThrowIfCancellationRequested();
119
120 using var activity = StartActivity();
121 var sw = System.Diagnostics.Stopwatch.StartNew();
122
123 try
124 {
125 await EnsureConnectedAsync(cancellationToken);
126
127 var existsMode = overwrite ? FtpRemoteExists.Overwrite : FtpRemoteExists.Skip;
128 var createDir = _options.AutoCreateDirectory ? FtpLocalExists.Overwrite : FtpLocalExists.Skip;
129
130 var status = await _client.UploadFile(
131 localPath,
132 remotePath,
133 existsMode,
134 _options.AutoCreateDirectory,
135 FtpVerify.None,
136 null,
137 cancellationToken);
138
139 if (status == FtpStatus.Failed)
140 {
141 throw new IOException($"Failed to upload file to {remotePath}");
142 }
143
144 var fileSize = new FileInfo(localPath).Length;
145
146 _logger.LogDebug(
147 "Uploaded {LocalPath} to {RemotePath} (status: {Status})",
148 localPath,
149 remotePath,
150 status);
151
152 RecordOperation("UploadOne", sw.ElapsedMilliseconds);
153 ToolboxMeter.RecordFileUpload(ServiceName, _options.Protocol.ToString(), _options.Host, fileSize);
154 }
155 catch (Exception ex) when (ex is not ArgumentNullException and not FileNotFoundException and not OperationCanceledException and not ObjectDisposedException)
156 {
157 _logger.LogError(ex, "Failed to upload {LocalPath} to {RemotePath}", localPath, remotePath);
158 ToolboxMeter.RecordFileTransferError(ServiceName, _options.Protocol.ToString(), _options.Host, "upload", ex.GetType().Name);
159 throw new IOException($"Failed to upload file: {ex.Message}", ex);
160 }

170 public async Task<int> UploadBatchAsync(IEnumerable<(string LocalPath, string RemotePath)> files, bool overwrite = true, CancellationToken cancellationToken = default)
171 {
172 ThrowIfDisposed();
173 ArgumentNullException.ThrowIfNull(files);
174
175 cancellationToken.ThrowIfCancellationRequested();
176
177 using var activity = StartActivity();
178 var sw = System.Diagnostics.Stopwatch.StartNew();
179
180 var fileList = files.ToList();
181 var successCount = 0;
182 var exceptions = new List<Exception>();
183
184 await EnsureConnectedAsync(cancellationToken);
185
186 foreach (var (localPath, remotePath) in fileList)
187 {
188 try
189 {
190 await UploadOneAsync(localPath, remotePath, overwrite, cancellationToken);
191 successCount++;
192 }
193 catch (Exception ex) when (ex is not OperationCanceledException)
194 {
195 exceptions.Add(new IOException($"Failed to upload {localPath}: {ex.Message}", ex));
196 }
197 }
198
199 RecordOperation("UploadBatch", sw.ElapsedMilliseconds);
200
201 if (exceptions.Count > 0)
202 {
203 _logger.LogWarning(
204 "Batch upload completed with {SuccessCount}/{TotalCount} successes and {ErrorCount} errors",
205 successCount,
206 fileList.Count,
207 exceptions.Count);
208
209 if (successCount == 0)
210 {
211 throw new AggregateException("All file uploads failed.", exceptions);
212 }
213 }
214
215 return successCount;

225 public async Task DownloadOneAsync(string remotePath, string localPath, bool overwrite = true, CancellationToken cancellationToken = default)
226 {
227 ThrowIfDisposed();
228 ArgumentNullException.ThrowIfNull(remotePath);
229 ArgumentNullException.ThrowIfNull(localPath);
230
231 if (!overwrite && File.Exists(localPath))
232 {
233 _logger.LogDebug("Skipping download, local file exists: {LocalPath}", localPath);
234 return;
235 }
236
237 cancellationToken.ThrowIfCancellationRequested();
238
239 using var activity = StartActivity();
240 var sw = System.Diagnostics.Stopwatch.StartNew();
241
242 try
243 {
244 await EnsureConnectedAsync(cancellationToken);
245
246 // Ensure local directory exists
247 var localDir = Path.GetDirectoryName(localPath);
248 if (!string.IsNullOrEmpty(localDir) && !Directory.Exists(localDir))
249 {
250 Directory.CreateDirectory(localDir);
251 }
252
253 var existsMode = overwrite ? FtpLocalExists.Overwrite : FtpLocalExists.Skip;
254
255 var status = await _client.DownloadFile(
256 localPath,
257 remotePath,
258 existsMode,
259 FtpVerify.None,
260 null,
261 cancellationToken);
262
263 if (status == FtpStatus.Failed)
264 {
265 throw new IOException($"Failed to download file from {remotePath}");
266 }
267
268 var fileSize = new FileInfo(localPath).Length;
269
270 _logger.LogDebug(
271 "Downloaded {RemotePath} to {LocalPath} (status: {Status})",
272 remotePath,
273 localPath,
274 status);
275
276 RecordOperation("DownloadOne", sw.ElapsedMilliseconds);
277 ToolboxMeter.RecordFileDownload(ServiceName, _options.Protocol.ToString(), _options.Host, fileSize);
278 }
279 catch (Exception ex) when (ex.Message.Contains("not found", StringComparison.OrdinalIgnoreCase) ||
280 ex.Message.Contains("doesn't exist", StringComparison.OrdinalIgnoreCase) ||
281 ex.Message.Contains("550", StringComparison.OrdinalIgnoreCase))
282 {
283 throw new FileNotFoundException($"Remote file not found: {remotePath}", remotePath, ex);
284 }
285 catch (Exception ex) when (ex is not ArgumentNullException and not FileNotFoundException and not OperationCanceledException and not ObjectDisposedException)
286 {
287 _logger.LogError(ex, "Failed to download {RemotePath} to {LocalPath}", remotePath, localPath);
288 ToolboxMeter.RecordFileTransferError(ServiceName, _options.Protocol.ToString(), _options.Host, "download", ex.GetType().Name);
289 throw new IOException($"Failed to download file: {ex.Message}", ex);
290 }

391 private async Task EnsureConnectedAsync(CancellationToken cancellationToken)
392 {
393 if (_isConnected && _client.IsConnected)
394 {
395 return;
396 }
397
398 await _connectionLock.WaitAsync(cancellationToken);
399 try
400 {
401 if (!_isConnected || !_client.IsConnected)
402 {
403 await _client.Connect(cancellationToken);
404 _isConnected = true;
405 _logger.LogDebug("Connected to FTP server {Host}:{Port}", _options.Host, _options.GetEffectivePort());
406 }
