121 X509Certificate2 certificate,
122 RsaPaddingMode paddingMode,
123 ILogger<RsaCryptographyService> logger)

125 {
126 ArgumentNullException.ThrowIfNull(certificate);
127 ArgumentNullException.ThrowIfNull(logger);
128
129 _logger = logger;
130 _padding = GetEncryptionPadding(paddingMode);
131

231 public static RsaKeyPair GenerateCertificate(
232 RsaKeySize keySize,
233 string subjectName,
234 TimeSpan validityPeriod,
235 bool includePrivateKey = true)
236 {
237 ArgumentNullException.ThrowIfNull(subjectName);
238 ValidateKeySize((int)keySize);
239
240 using var rsa = RSA.Create((int)keySize);
241
242 var request = new CertificateRequest(
243 subjectName,
244 rsa,
245 HashAlgorithmName.SHA256,
246 RSASignaturePadding.Pkcs1);
247
248 var notBefore = DateTimeOffset.UtcNow;
249 var notAfter = notBefore.Add(validityPeriod);
250
251 using var certWithPrivateKey = request.CreateSelfSigned(notBefore, notAfter);
252
253 X509Certificate2 resultCert;
254 if (includePrivateKey)
255 {
256 // Export and re-import to ensure private key is exportable
257 var pfxBytes = certWithPrivateKey.Export(X509ContentType.Pfx);
258 resultCert = X509CertificateLoader.LoadPkcs12(pfxBytes, null, X509KeyStorageFlags.Exportable);
259 }
260 else
261 {
262 // Export only the public certificate
263 var certBytes = certWithPrivateKey.Export(X509ContentType.Cert);
264 resultCert = X509CertificateLoader.LoadCertificate(certBytes);
265 }
266
267 return RsaKeyPair.FromCertificate(resultCert);
268 }

294 public string Encrypt(string plainText)
295 {
296 ThrowIfDisposed();
297 ArgumentNullException.ThrowIfNull(plainText);
298
299 using var activity = StartActivity();
300 var sw = System.Diagnostics.Stopwatch.StartNew();
301
302 try
303 {
304 var plainBytes = Encoding.UTF8.GetBytes(plainText);
305 var maxDataLength = GetMaxDataLength();
306
307 if (plainBytes.Length > maxDataLength)
308 {
309 throw new CryptographicException(
310 $"Data too large for RSA encryption. Maximum size is {maxDataLength} bytes, but data is {plainBytes.Length} bytes. " +
311 "Consider using hybrid encryption (RSA + AES) for larger data.");
312 }
313
314 var encryptedBytes = _rsa.Encrypt(plainBytes, _padding);
315 var result = Convert.ToBase64String(encryptedBytes);
316
317 _logger.LogDebug(
318 "Encrypted {InputLength} bytes to {OutputLength} characters",
319 plainBytes.Length,
320 result.Length);
321
322 RecordOperation("Encrypt", sw.ElapsedMilliseconds);
323 ToolboxMeter.RecordEncryption(ServiceName, "RSA", plainBytes.Length, _rsa.KeySize);
324 return result;
325 }
326 catch (Exception ex) when (ex is not ArgumentNullException and not CryptographicException)
327 {
328 _logger.LogError(ex, "Failed to encrypt text");
329 throw new CryptographicException("RSA encryption failed.", ex);
330 }
331 }

348 public string Decrypt(string encryptedText)
349 {
350 ThrowIfDisposed();
351 ArgumentNullException.ThrowIfNull(encryptedText);
352
353 if (!_hasPrivateKey)
354 {
355 throw new InvalidOperationException(
356 "Cannot decrypt without a private key. Ensure the service was initialized with a private key.");
357 }
358
359 using var activity = StartActivity();

361
362 try
363 {
364 var encryptedBytes = Convert.FromBase64String(encryptedText);

368 _logger.LogDebug(

370 encryptedText.Length,

372
373 RecordOperation("Decrypt", sw.ElapsedMilliseconds);

420 RsaPaddingMode.Pkcs1 => RSAEncryptionPadding.Pkcs1,

447 private int GetMaxDataLength()
448 {
449 var keyBytes = _rsa.KeySize / 8;
450
451 // OAEP overhead depends on hash size
452 // Formula: keyBytes - 2 * hashBytes - 2
453 return _padding.OaepHashAlgorithm.Name switch
454 {
455 "SHA1" => keyBytes - 42, // 2 * 20 + 2 = 42
456 "SHA256" => keyBytes - 66, // 2 * 32 + 2 = 66
457 "SHA384" => keyBytes - 98, // 2 * 48 + 2 = 98
458 "SHA512" => keyBytes - 130, // 2 * 64 + 2 = 130
459 _ => keyBytes - 11 // PKCS#1 v1.5 overhead is 11 bytes
460 };
