282 public string BuildUrl()
283 {
284 if (QueryParameters.Count == 0)
285 {
286 return Url;
287 }
288
289 var separator = Url.Contains('?') ? "&" : "?";
290 var queryString = string.Join("&",
291 QueryParameters.Select(kvp =>
292 $"{Uri.EscapeDataString(kvp.Key)}={Uri.EscapeDataString(kvp.Value)}"));
293
294 return $"{Url}{separator}{queryString}";
295 }
