109 public async Task SendMailAsync(EmailMessage message, CancellationToken cancellationToken = default)
110 {
111 ThrowIfDisposed();
112 ArgumentNullException.ThrowIfNull(message);
113
114 message.Validate();
115
116 cancellationToken.ThrowIfCancellationRequested();
117
118 using var activity = StartActivity();
119 var sw = System.Diagnostics.Stopwatch.StartNew();
120
121 try
122 {
123 await EnsureConnectedAsync(cancellationToken);
124
125 var mimeMessage = BuildMimeMessage(message);
126
127 await _client.SendAsync(mimeMessage, cancellationToken);
128
129 _logger.LogDebug(
130 "Email sent successfully to {RecipientCount} recipient(s): Subject='{Subject}'",
131 message.TotalRecipients,
132 message.Subject);
133
134 RecordOperation("SendMail", sw.ElapsedMilliseconds);
135 RecordEmailSent(message);
136 }
137 catch (SmtpCommandException ex)
138 {
139 _logger.LogError(
140 ex,
141 "SMTP command error while sending email: {StatusCode} - {Message}",
142 ex.StatusCode,
143 ex.Message);
144
145 throw new InvalidOperationException($"SMTP error: {ex.Message}", ex);
146 }
147 catch (SmtpProtocolException ex)
148 {
149 _logger.LogError(ex, "SMTP protocol error while sending email");
150 throw new InvalidOperationException($"SMTP protocol error: {ex.Message}", ex);
151 }
152 catch (Exception ex) when (ex is not ArgumentNullException
153 and not ArgumentException
154 and not OperationCanceledException
155 and not ObjectDisposedException
156 and not InvalidOperationException)
157 {
158 _logger.LogError(ex, "Failed to send email");
159 throw new InvalidOperationException($"Failed to send email: {ex.Message}", ex);
160 }
161 }

189 private MimeMessage BuildMimeMessage(EmailMessage message)
190 {
191 var mimeMessage = new MimeMessage();
192
193 // From address
194 var from = message.From ?? _options.DefaultFrom
195 ?? throw new ArgumentException("No sender address specified and no default configured.");
196
197 mimeMessage.From.Add(new MailboxAddress(from.DisplayName, from.Address));
198
199 // Reply-To
200 var replyTo = message.ReplyTo ?? _options.DefaultReplyTo;
201 if (replyTo is not null)
202 {
203 mimeMessage.ReplyTo.Add(new MailboxAddress(replyTo.DisplayName, replyTo.Address));
204 }
205
206 // Recipients
207 foreach (var to in message.To)
208 {
209 mimeMessage.To.Add(new MailboxAddress(to.DisplayName, to.Address));
210 }
211
212 foreach (var cc in message.Cc)
213 {
214 mimeMessage.Cc.Add(new MailboxAddress(cc.DisplayName, cc.Address));
215 }
216
217 foreach (var bcc in message.Bcc)
218 {
219 mimeMessage.Bcc.Add(new MailboxAddress(bcc.DisplayName, bcc.Address));
220 }
221
222 // Subject
223 mimeMessage.Subject = message.Subject;
224
225 // Priority
226 mimeMessage.Priority = message.Priority switch
227 {
228 Options.EmailPriority.Low => MessagePriority.NonUrgent,
229 Options.EmailPriority.High => MessagePriority.Urgent,
230 _ => MessagePriority.Normal
231 };
232
233 // Custom headers
234 foreach (var (name, value) in message.Headers)
235 {
236 mimeMessage.Headers.Add(name, value);
237 }
238
239 // Body and attachments
240 mimeMessage.Body = BuildMessageBody(message);
241
242 return mimeMessage;
243 }

250 private static MimeEntity BuildMessageBody(EmailMessage message)
251 {
252 // Create the body part
253 var textFormat = message.IsBodyHtml ? TextFormat.Html : TextFormat.Plain;
254 var textPart = new TextPart(textFormat)
255 {
256 Text = message.Body
257 };
258
259 // If HTML with explicit plain text, create multipart/alternative
260 MimeEntity bodyPart;
261 if (message.IsBodyHtml && !string.IsNullOrEmpty(message.PlainTextBody))
262 {
263 var alternative = new Multipart("alternative")
264 {
265 new TextPart(TextFormat.Plain) { Text = message.PlainTextBody },
266 textPart
267 };
268 bodyPart = alternative;
269 }
270 else
271 {
272 bodyPart = textPart;
273 }
274
275 // Handle attachments
276 if (message.Attachments.Count == 0)
277 {
278 return bodyPart;
279 }
280
281 // Separate inline and regular attachments
282 var inlineAttachments = message.Attachments.Where(a => a.IsInline).ToList();
283 var regularAttachments = message.Attachments.Where(a => !a.IsInline).ToList();
284
285 // If we have inline attachments and HTML body, wrap in multipart/related
286 if (inlineAttachments.Count > 0 && message.IsBodyHtml)
287 {
288 var related = new Multipart("related") { bodyPart };
289
290 foreach (var attachment in inlineAttachments)
291 {
292 related.Add(CreateMimePart(attachment));
293 }
294
295 bodyPart = related;
296 }
297
298 // If we have regular attachments, wrap in multipart/mixed
299 if (regularAttachments.Count > 0)
300 {
301 var mixed = new Multipart("mixed") { bodyPart };
302
303 foreach (var attachment in regularAttachments)
304 {
305 mixed.Add(CreateMimePart(attachment));
306 }
307
308 return mixed;
309 }
310
311 return bodyPart;
312 }

319 private static MimePart CreateMimePart(EmailAttachment attachment)
320 {
321 var contentType = ContentType.Parse(attachment.ContentType);
322 var mimePart = new MimePart(contentType)
323 {
324 FileName = attachment.FileName,
325 ContentDisposition = new ContentDisposition(
326 attachment.IsInline
327 ? ContentDisposition.Inline
328 : ContentDisposition.Attachment),
329 ContentTransferEncoding = ContentEncoding.Base64
330 };
331
332 if (attachment.IsInline && !string.IsNullOrEmpty(attachment.ContentId))
333 {
334 mimePart.ContentId = attachment.ContentId;
335 }
336
337 // Set content
338 if (attachment.Content is not null)
339 {
340 mimePart.Content = new MimeContent(new MemoryStream(attachment.Content));
341 }
342 else if (!string.IsNullOrEmpty(attachment.FilePath))
343 {
344 mimePart.Content = new MimeContent(File.OpenRead(attachment.FilePath));
345 }
346
347 return mimePart;

354 private async Task EnsureConnectedAsync(CancellationToken cancellationToken)
355 {
356 if (_isConnected && _client.IsConnected)
357 {
358 return;
359 }
360
361 await _connectionLock.WaitAsync(cancellationToken);
362 try
363 {
364 if (_isConnected && _client.IsConnected)
365 {
366 return;
367 }
368
369 // Convert security mode to SecureSocketOptions
370 var secureSocketOptions = _options.SecurityMode switch
371 {
372 SmtpSecurityMode.None => SecureSocketOptions.None,
373 SmtpSecurityMode.StartTls => SecureSocketOptions.StartTls,
374 SmtpSecurityMode.StartTlsWhenAvailable => SecureSocketOptions.StartTlsWhenAvailable,
375 SmtpSecurityMode.SslOnConnect => SecureSocketOptions.SslOnConnect,
376 _ => SecureSocketOptions.Auto
377 };
378
379 await _client.ConnectAsync(
380 _options.Host,
381 _options.Port,
382 secureSocketOptions,
383 cancellationToken);
384
385 _logger.LogDebug(
386 "Connected to SMTP server {Host}:{Port}",
387 _options.Host,
388 _options.Port);
389
390 // Authenticate if credentials are configured
391 if (_options.RequiresAuthentication)
392 {
393 await AuthenticateAsync(cancellationToken);
394 }
395
396 _isConnected = true;
397 }
398 catch (Exception ex) when (ex is not OperationCanceledException)
399 {
400 _logger.LogError(ex, "Failed to connect to SMTP server {Host}:{Port}", _options.Host, _options.Port);
401 throw new InvalidOperationException($"Failed to connect to SMTP server: {ex.Message}", ex);
402 }
403 finally
404 {
405 _connectionLock.Release();
406 }
407 }

413 private async Task AuthenticateAsync(CancellationToken cancellationToken)
414 {
415 try
416 {
417 if (!string.IsNullOrEmpty(_options.OAuth2AccessToken))
418 {
419 // OAuth2 authentication
420 var oauth2 = new SaslMechanismOAuth2(_options.Username!, _options.OAuth2AccessToken);
421 await _client.AuthenticateAsync(oauth2, cancellationToken);
422
423 _logger.LogDebug("Authenticated with SMTP server using OAuth2");
424 }
425 else if (!string.IsNullOrEmpty(_options.Username))
426 {
427 // Password authentication
428 await _client.AuthenticateAsync(
429 _options.Username,
430 _options.Password ?? string.Empty,
431 cancellationToken);
432
433 _logger.LogDebug("Authenticated with SMTP server using password");
434 }
435 }
436 catch (AuthenticationException ex)
437 {
438 _logger.LogError(ex, "SMTP authentication failed");
439 throw new InvalidOperationException($"SMTP authentication failed: {ex.Message}", ex);
440 }
