111 public string Encrypt(string plainText)
112 {
113 ThrowIfDisposed();
114 ArgumentNullException.ThrowIfNull(plainText);
115
116 using var activity = StartActivity();
117 var sw = System.Diagnostics.Stopwatch.StartNew();
118
119 try
120 {
121 using var aes = Aes.Create();
122 aes.Key = _key;
123 aes.IV = _iv;
124 aes.Mode = CipherMode.CBC;
125 aes.Padding = PaddingMode.PKCS7;
126
127 using var encryptor = aes.CreateEncryptor();
128 var plainBytes = Encoding.UTF8.GetBytes(plainText);
129 var encryptedBytes = encryptor.TransformFinalBlock(plainBytes, 0, plainBytes.Length);
130 var result = Convert.ToBase64String(encryptedBytes);
131
132 _logger.LogDebug(
133 "Encrypted {InputLength} characters to {OutputLength} characters",
134 plainText.Length,
135 result.Length);
136
137 RecordOperation("Encrypt", sw.ElapsedMilliseconds);
138 ToolboxMeter.RecordEncryption(ServiceName, "AES", plainBytes.Length, _key.Length * 8);
139 return result;
140 }
141 catch (Exception ex) when (ex is not ArgumentNullException)
142 {
143 _logger.LogError(ex, "Failed to encrypt text");
144 throw;
