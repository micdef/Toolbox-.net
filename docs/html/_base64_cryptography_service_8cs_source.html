181 private string EncodeToBase64(byte[] bytes)
182 {
183 var base64 = Convert.ToBase64String(bytes);
184
185 // Convert to URL-safe if needed
186 if (_encodingTable == Base64EncodingTable.UrlSafe)
187 {
188 base64 = base64.Replace('+', '-').Replace('/', '_');
189 }
190
191 // Remove padding if configured
192 if (!_includePadding)
193 {
194 base64 = base64.TrimEnd('=');
195 }
196
197 return base64;
198 }

206 private byte[] DecodeFromBase64(string base64)
207 {
208 // Convert from URL-safe if needed
209 if (_encodingTable == Base64EncodingTable.UrlSafe)
210 {
211 base64 = base64.Replace('-', '+').Replace('_', '/');
212 }
213
214 // Add padding if missing
215 var paddingNeeded = base64.Length % 4;
216 if (paddingNeeded > 0)
217 {
218 base64 += new string('=', 4 - paddingNeeded);
219 }
220
221 return Convert.FromBase64String(base64);
