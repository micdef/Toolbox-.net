123 public async Task<ApiResponse> SendAsync(ApiRequest request, CancellationToken cancellationToken = default)
124 {
125 ThrowIfDisposed();
126 ArgumentNullException.ThrowIfNull(request);
127
128 cancellationToken.ThrowIfCancellationRequested();
129
130 using var activity = StartActivity();
131 var sw = System.Diagnostics.Stopwatch.StartNew();
132
133 var retryCount = 0;
134 HttpResponseMessage? response = null;
135 Exception? lastException = null;
136
137 while (retryCount <= _options.MaxRetries)
138 {
139 try
140 {
141 using var httpRequest = await BuildHttpRequestAsync(request, cancellationToken);
142
143 _logger.LogDebug(
144 "Sending {Method} request to {Url}",
145 request.Method,
146 httpRequest.RequestUri);
147
148 response = await _httpClient.SendAsync(httpRequest, cancellationToken);
149
150 // Don't retry on success or client errors
151 if (response.IsSuccessStatusCode || (int)response.StatusCode < 500)
152 {
153 break;
154 }
155
156 lastException = new HttpRequestException(
157 $"Server error: {response.StatusCode}",
158 null,
159 response.StatusCode);
160 }
161 catch (HttpRequestException ex)
162 {
163 lastException = ex;
164 }
165 catch (TaskCanceledException ex) when (!cancellationToken.IsCancellationRequested)
166 {
167 lastException = new HttpRequestException("Request timed out", ex);
168 }
169
170 retryCount++;
171
172 if (retryCount <= _options.MaxRetries)
173 {
174 var delay = _options.UseExponentialBackoff
175 ? TimeSpan.FromMilliseconds(_options.RetryDelay.TotalMilliseconds * Math.Pow(2, retryCount - 1))
176 : _options.RetryDelay;
177
178 _logger.LogWarning(
179 "Request failed, retrying in {Delay}ms (attempt {Attempt}/{MaxRetries})",
180 delay.TotalMilliseconds,
181 retryCount,
182 _options.MaxRetries);
183
184 await Task.Delay(delay, cancellationToken);
185 }
186 }
187
188 if (response is null)
189 {
190 throw lastException ?? new HttpRequestException("Request failed after all retries");
191 }
192
193 var apiResponse = await BuildApiResponseAsync(response, sw.Elapsed, cancellationToken);
194
195 RecordOperation("Send", sw.ElapsedMilliseconds);
196 RecordApiRequest(request, apiResponse);
197
198 _logger.LogDebug(
199 "Received {StatusCode} response in {Duration}ms",
200 apiResponse.StatusCode,
201 sw.ElapsedMilliseconds);
202
203 return apiResponse;
204 }

238 private HttpClient CreateHttpClient()
239 {
240 var handler = new HttpClientHandler
241 {
242 AllowAutoRedirect = _options.FollowRedirects,
243 MaxAutomaticRedirections = _options.MaxRedirects
244 };
245
246 // SSL certificate validation
247 if (!_options.ValidateCertificate)
248 {
249 handler.ServerCertificateCustomValidationCallback = (_, _, _, _) => true;
250 }
251
252 // Client certificate authentication
253 if (_options.AuthenticationMode == ApiAuthenticationMode.Certificate)
254 {
255 var certificate = GetClientCertificate();
256 if (certificate is not null)
257 {
258 handler.ClientCertificates.Add(certificate);
259 }
260 }
261
262 var client = new HttpClient(handler)
263 {
264 Timeout = _options.Timeout
265 };
266
267 ConfigureHttpClient(client);
268
269 return client;

325 private async Task<HttpRequestMessage> BuildHttpRequestAsync(
326 ApiRequest request,
327 CancellationToken cancellationToken)
328 {
329 var url = request.BuildUrl();
330
331 // Handle API key in query string
332 if (_options.AuthenticationMode == ApiAuthenticationMode.ApiKey &&
333 _options.ApiKeyLocation == ApiKeyLocation.QueryString &&
334 !string.IsNullOrEmpty(_options.ApiKey))
335 {
336 var separator = url.Contains('?') ? "&" : "?";
337 url = $"{url}{separator}{Uri.EscapeDataString(_options.ApiKeyName)}={Uri.EscapeDataString(_options.ApiKey)}";
338 }
339
340 var httpRequest = new HttpRequestMessage(new HttpMethod(request.Method), url)
341 {
342 Content = request.Content
343 };
344
345 // Add request headers
346 foreach (var (name, value) in request.Headers)
347 {
348 httpRequest.Headers.TryAddWithoutValidation(name, value);
349 }
350
351 // Add authentication
352 await AddAuthenticationAsync(httpRequest, cancellationToken);
353
354 return httpRequest;
355 }

362 private async Task AddAuthenticationAsync(HttpRequestMessage request, CancellationToken cancellationToken)
363 {
364 switch (_options.AuthenticationMode)
365 {
366 case ApiAuthenticationMode.BearerToken:
367 if (!string.IsNullOrEmpty(_options.BearerToken))
368 {
369 request.Headers.Authorization = new AuthenticationHeaderValue("Bearer", _options.BearerToken);
370 }
371 break;
372
373 case ApiAuthenticationMode.Basic:
374 if (!string.IsNullOrEmpty(_options.Username))
375 {
376 var credentials = Convert.ToBase64String(
377 Encoding.ASCII.GetBytes($"{_options.Username}:{_options.Password ?? string.Empty}"));
378 request.Headers.Authorization = new AuthenticationHeaderValue("Basic", credentials);
379 }
380 break;
381
382 case ApiAuthenticationMode.ApiKey:
383 if (_options.ApiKeyLocation == ApiKeyLocation.Header && !string.IsNullOrEmpty(_options.ApiKey))
384 {
385 request.Headers.TryAddWithoutValidation(_options.ApiKeyName, _options.ApiKey);
386 }
387 break;
388
389 case ApiAuthenticationMode.OAuth2ClientCredentials:
390 var token = await GetOAuth2TokenAsync(cancellationToken);
391 if (!string.IsNullOrEmpty(token))
392 {
393 request.Headers.Authorization = new AuthenticationHeaderValue("Bearer", token);
394 }
395 break;
396
397 case ApiAuthenticationMode.Certificate:
398 // Certificate is added at the handler level
399 break;
400
401 case ApiAuthenticationMode.Anonymous:
402 default:
403 // No authentication
404 break;
405 }

413 private async Task<string?> GetOAuth2TokenAsync(CancellationToken cancellationToken)
414 {
415 // Check cached token
416 if (!string.IsNullOrEmpty(_oauth2Token) && DateTime.UtcNow < _oauth2TokenExpiry)
417 {
418 return _oauth2Token;
419 }
420
421 await _oauth2Lock.WaitAsync(cancellationToken);
422 try
423 {
424 // Double-check after acquiring lock
425 if (!string.IsNullOrEmpty(_oauth2Token) && DateTime.UtcNow < _oauth2TokenExpiry)
426 {
427 return _oauth2Token;
428 }
429
430 if (string.IsNullOrEmpty(_options.OAuth2TokenUrl) ||
431 string.IsNullOrEmpty(_options.OAuth2ClientId) ||
432 string.IsNullOrEmpty(_options.OAuth2ClientSecret))
433 {
434 _logger.LogWarning("OAuth2 credentials not configured");
435 return null;
436 }
437
438 using var tokenClient = new HttpClient();
439
440 var tokenRequest = new Dictionary<string, string>
441 {
442 ["grant_type"] = "client_credentials",
443 ["client_id"] = _options.OAuth2ClientId,
444 ["client_secret"] = _options.OAuth2ClientSecret
445 };
446
447 if (!string.IsNullOrEmpty(_options.OAuth2Scopes))
448 {
449 tokenRequest["scope"] = _options.OAuth2Scopes;
450 }
451
452 using var content = new FormUrlEncodedContent(tokenRequest);
453 using var response = await tokenClient.PostAsync(_options.OAuth2TokenUrl, content, cancellationToken);
454
455 response.EnsureSuccessStatusCode();
456
457 var json = await response.Content.ReadAsStringAsync(cancellationToken);
458 using var doc = JsonDocument.Parse(json);
459
460 _oauth2Token = doc.RootElement.GetProperty("access_token").GetString();
461
462 if (doc.RootElement.TryGetProperty("expires_in", out var expiresIn))
463 {
464 var expiresInSeconds = expiresIn.GetInt32();
465 _oauth2TokenExpiry = DateTime.UtcNow.AddSeconds(expiresInSeconds - 60); // 1 minute buffer
466 }
467 else
468 {
469 _oauth2TokenExpiry = DateTime.UtcNow.AddMinutes(55); // Default 55 minutes
470 }
471
472 _logger.LogDebug("Obtained OAuth2 token, expires at {Expiry}", _oauth2TokenExpiry);
473
474 return _oauth2Token;
475 }
476 finally
477 {
478 _oauth2Lock.Release();
479 }
480 }

489 private static async Task<ApiResponse> BuildApiResponseAsync(
490 HttpResponseMessage response,
491 TimeSpan duration,
492 CancellationToken cancellationToken)
493 {
494 var body = await response.Content.ReadAsStringAsync(cancellationToken);
495 var bodyBytes = await response.Content.ReadAsByteArrayAsync(cancellationToken);
496
497 var headers = new Dictionary<string, IEnumerable<string>>();
498 foreach (var header in response.Headers)
499 {
500 headers[header.Key] = header.Value;
501 }
502 foreach (var header in response.Content.Headers)
503 {
504 headers[header.Key] = header.Value;
505 }
506
507 return new ApiResponse
508 {
509 StatusCode = response.StatusCode,
510 ReasonPhrase = response.ReasonPhrase,
511 Body = body,
512 BodyBytes = bodyBytes,
513 Headers = headers,
514 ContentType = response.Content.Headers.ContentType?.ToString(),
515 ContentLength = response.Content.Headers.ContentLength,
516 Duration = duration
517 };
518 }
